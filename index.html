class DevFriend extends Friend {

	//same super.constructor(name, city)

function createArrayIterator(arr) {
    let index = 0;
    return {
        next() {
            if (index < arr.length) {
                return { done: false, value: arr[index++] };
            } else {
                return { done: true };
            }
        }
    }
}
